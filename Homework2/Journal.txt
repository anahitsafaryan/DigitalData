# Journal

authors 
*id integer auto_increment
-name varchar(50) 
-surname varchar (50) 

reviewers 
*id integer auto_increment
-name varchar (50) 
-surname varchar (50) 

issues 
*id integer auto_increment
-date? date

articles 
*id integer auto_increment
-name varchar (150) 
-author => authors.id
-reviewer => reviewers.id
-accepted tinyint(1)
-issue => issues.id
